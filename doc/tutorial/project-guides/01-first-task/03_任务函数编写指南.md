# 03 - 任务函数编写指南

## 任务函数的标准格式

### 基本结构

```c
void vTaskFunction(void *pvParameters)
{
    // 1. 初始化区域（可选，只运行一次）
    初始化硬件();
    初始化变量();
    
    // 2. 必须有无限循环
    for(;;)  // 或者 while(1)
    {
        // 3. 任务主体代码
        处理业务逻辑();
        
        // 4. 必须让出CPU（关键！）
        vTaskDelay(100);  // 或者等待信号量、队列等
    }
    
    // 5. 永远不应该到达这里！
    // 如果到这里，任务会被调度器删除
}
```

---

## 规范详解

### 1. 函数命名

**规范**：
- 使用FreeRTOS风格：前缀 `v` 表示返回 `void`
- 使用描述性名称，说明任务做什么
- 大驼峰命名法

**示例**：
```c
// ✅ 好的命名
void vLEDTask(void *pvParameters);        // LED控制任务
void vUARTReceiveTask(void *pvParameters); // 串口接收任务
void vSensorReadTask(void *pvParameters);  // 传感器读取任务

// ❌ 不好的命名
void task1(void *pvParameters);     // 不知道做什么
void func(void *pvParameters);       // 太通用
void my_task(void *pvParameters);    // 风格不一致
```

---

### 2. 参数处理

```c
void vMyTask(void *pvParameters)
{
    // 方法1：如果不使用参数，加上这行防止编译器警告
    (void)pvParameters;
    
    // 方法2：如果使用参数，转换为正确类型
    int *pMyData = (int *)pvParameters;
    
    // 任务代码...
}
```

---

### 3. 无限循环

**为什么必须是无限循环？**

- 任务就像一个工人，不能干完一件事就消失
- 任务应该持续工作，除非被故意删除
- 如果任务函数返回，调度器会认为任务结束了，自动删除它

**两种写法都可以**：
```c
// 写法1：for(;;)
for(;;)
{
    // 任务代码
}

// 写法2：while(1)
while(1)
{
    // 任务代码
}
```

推荐用 `for(;;)`，因为：
- 有些编译器对 `for(;;)` 优化更好
- FreeRTOS官方示例常用 `for(;;)`

---

### 4. 让出CPU

**关键原则**：任务必须定期让出CPU！

**错误做法**：
```c
void vBadTask(void *pvParameters)
{
    for(;;)
    {
        // ❌ 死循环，不让出CPU！
        // 其他任务永远无法运行
        做某事();
    }
}
```

**正确做法**：
```c
void vGoodTask(void *pvParameters)
{
    for(;;)
    {
        做某事();
        
        // ✅ 让出CPU
        vTaskDelay(10);  // 方法1：延时
        // 或者
        xQueueReceive(xQueue, &data, portMAX_DELAY);  // 方法2：等待队列
        // 或者
        xSemaphoreTake(xSemaphore, portMAX_DELAY);  // 方法3：等待信号量
    }
}
```

## 完整示例

### 示例1：简单的LED闪烁任务

```c
/**
 * @brief LED闪烁任务
 * @param pvParameters 未使用
 */
void vLEDTask(void *pvParameters)
{
    // 防止未使用参数警告
    (void)pvParameters;
    
    // 初始化LED（只运行一次）
    vLEDInit();
    
    for(;;)
    {
        // 点亮LED
        vLEDOn();
        
        // 延时500毫秒
        vTaskDelay(pdMS_TO_TICKS(500));
        
        // 熄灭LED
        vLEDOff();
        
        // 再延时500毫秒
        vTaskDelay(pdMS_TO_TICKS(500));
    }
}
```

---

### 示例2：带参数的任务

```c
// 定义参数结构体
typedef struct
{
    uint8_t ucLEDNumber;    // LED编号
    uint32_t ulDelayMs;     // 闪烁间隔（毫秒）
} LEDParams_t;

/**
 * @brief 通用LED控制任务
 * @param pvParameters 指向LEDParams_t的指针
 */
void vLEDBlinkTask(void *pvParameters)
{
    // 转换参数类型
    LEDParams_t *pxParams = (LEDParams_t *)pvParameters;
    
    // 初始化指定的LED
    vLEDInit(pxParams->ucLEDNumber);
    
    for(;;)
    {
        vLEDOn(pxParams->ucLEDNumber);
        vTaskDelay(pdMS_TO_TICKS(pxParams->ulDelayMs));
        
        vLEDOff(pxParams->ucLEDNumber);
        vTaskDelay(pdMS_TO_TICKS(pxParams->ulDelayMs));
    }
}

// 使用示例
void main(void)
{
    // 创建两个不同的LED任务，使用不同参数
    static LEDParams_t xParams1 = {1, 500};  // LED1，500ms
    static LEDParams_t xParams2 = {2, 1000}; // LED2，1000ms
    
    xTaskCreate(vLEDBlinkTask, "LED1", 128, &xParams1, 1, NULL);
    xTaskCreate(vLEDBlinkTask, "LED2", 128, &xParams2, 1, NULL);
    
    vTaskStartScheduler();
}
```

### 示例3：串口接收任务

```c
/**
 * @brief 串口接收任务
 * @param pvParameters 未使用
 */
void vUARTReceiveTask(void *pvParameters)
{
    (void)pvParameters;
    
    uint8_t ucRxData;
    
    for(;;)
    {
        // 等待串口数据（如果没有数据，任务会阻塞在这里）
        if(xUARTReceive(&ucRxData, portMAX_DELAY) == pdPASS)
        {
            // 处理接收到的数据
            vProcessData(ucRxData);
        }
    }
}
```

## 常见错误

### 错误1：任务函数返回

```c
// ❌ 错误！
void vBadTask(void *pvParameters)
{
    做某事();
    // 没有无限循环！函数会返回
}

// 结果：任务运行一次就被删除了！
```

---

### 错误2：不让出CPU

```c
// ❌ 错误！
void vBadTask(void *pvParameters)
{
    for(;;)
    {
        // 死循环，没有延时或等待
        做某事();
    }
}

// 结果：其他任务无法运行！
```

---

### 错误3：栈溢出

```c
// ❌ 危险！
void vBadTask(void *pvParameters)
{
    // 局部数组太大！
    uint8_t ucBigBuffer[1024];  // 需要1KB栈空间
    
    for(;;)
    {
        // 使用缓冲区
    }
}

// 创建任务时栈给得太小
xTaskCreate(vBadTask, "BadTask", 128, ...);  // 只有512字节栈！

// 结果：栈溢出，系统崩溃！
```

**解决方法**：
```c
// ✅ 正确
void vGoodTask(void *pvParameters)
{
    // 方案1：使用静态全局变量
    static uint8_t ucBuffer[1024];  // 不占用栈空间
    
    // 方案2：增加任务栈大小
    // xTaskCreate(vGoodTask, "GoodTask", 512, ...);
    
    for(;;)
    {
        // 使用缓冲区
    }
}
```

## 最佳实践

### 1. 任务职责单一

**每个任务只做一件事**：
```c
// ✅ 好：一个任务只控制LED
void vLEDTask(void *pvParameters);

// ✅ 好：一个任务只处理串口
void vUARTTask(void *pvParameters);

// ❌ 不好：一个任务做太多事
void vDoEverythingTask(void *pvParameters);
```

---

### 2. 合理的延时时间

```c
// ✅ 好：根据实际需要选择延时
vTaskDelay(pdMS_TO_TICKS(10));   // 快速响应：10ms
vTaskDelay(pdMS_TO_TICKS(100));  // 普通：100ms
vTaskDelay(pdMS_TO_TICKS(1000)); // 慢速：1s

// ❌ 不好：延时太长或太短
vTaskDelay(pdMS_TO_TICKS(1));    // 太短，浪费CPU
vTaskDelay(pdMS_TO_TICKS(60000)); // 太长，响应太慢
```

---

### 3. 正确处理参数

```c
// ✅ 好：使用静态或全局变量
static TaskParams_t xParams;  // 任务运行期间一直有效

void main(void)
{
    xParams.value = 123;
    xTaskCreate(vTask, "Task", 128, &xParams, 1, NULL);
}

// ❌ 危险：使用局部变量
void main(void)
{
    TaskParams_t xParams;  // 局部变量！函数返回后失效
    xParams.value = 123;
    xTaskCreate(vTask, "Task", 128, &xParams, 1, NULL);
    // 函数返回后，xParams内存被释放，任务访问会出错！
}
```

---

### 4. 添加注释

```c
/**
 * @brief 传感器读取任务
 * @param pvParameters 指向SensorConfig_t的指针
 *
 * 这个任务负责定期读取传感器数据，
 * 并通过队列发送给处理任务。
 *
 * 读取间隔：100ms
 */
void vSensorReadTask(void *pvParameters)
{
    (void)pvParameters;
    
    SensorData_t xData;
    
    for(;;)
    {
        // 读取传感器数据
        // 注意：这里使用阻塞方式读取
        if(xSensorRead(&xData) == pdPASS)
        {
            // 发送到处理队列
            xQueueSend(xDataQueue, &xData, 0);
        }
        
        // 延时100ms再读下一次
        vTaskDelay(pdMS_TO_TICKS(100));
    }
}
```

## 总结

### 任务函数编写检查清单

- [ ] 函数格式：`void vFuncName(void *pvParameters)`
- [ ] 有无限循环：`for(;;)` 或 `while(1)`
- [ ] 循环内有让出CPU的操作：`vTaskDelay()` 或等待队列/信号量
- [ ] 不使用的参数用 `(void)pvParameters;` 避免警告
- [ ] 栈大小足够，不会溢出
- [ ] 任务职责单一，不做太多事情
- [ ] 有清晰的注释说明任务功能
