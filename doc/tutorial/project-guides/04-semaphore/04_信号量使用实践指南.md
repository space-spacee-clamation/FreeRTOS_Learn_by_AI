## 常见使用模式

### 模式1：一对一生产者-消费者

```c
// 一个生产者，一个消费者
QueueHandle_t xQueue;

// 生产者任务
void vProducerTask(void *p)
{
    for(;;)
    {
        uint32_t data = 获取数据();
        xQueueSend(xQueue, &data, portMAX_DELAY);
        vTaskDelay(100);
    }
}

// 消费者任务
void vConsumerTask(void *p)
{
    for(;;)
    {
        uint32_t data;
        xQueueReceive(xQueue, &data, portMAX_DELAY);
        处理数据(data);
    }
}
```

**特点**：
- 最简单的队列使用场景
- 生产者和消费者一对一
- 队列作为数据缓冲区

---

### 模式2：多生产者，单消费者

```c
QueueHandle_t xWorkQueue;

// 生产者1
void vProducer1(void *p)
{
    for(;;)
    {
        WorkItem_t item = 创建工作1();
        xQueueSend(xWorkQueue, &item, 0);
        vTaskDelay(200);
    }
}

// 生产者2
void vProducer2(void *p)
{
    for(;;)
    {
        WorkItem_t item = 创建工作2();
        xQueueSend(xWorkQueue, &item, 0);
        vTaskDelay(300);
    }
}

// 消费者（工作处理者）
void vWorkerTask(void *p)
{
    for(;;)
    {
        WorkItem_t item;
        xQueueReceive(xWorkQueue, &item, portMAX_DELAY);
        处理工作(item);
    }
}
```

**特点**：
- 多个任务产生数据
- 一个任务集中处理
- 适合工作队列模式

### 模式3：单生产者，多消费者

```c
QueueHandle_t xDataQueue;

// 生产者
void vDataSourceTask(void *p)
{
    for(;;)
    {
        Data_t data = 采集数据();
        xQueueSend(xDataQueue, &data, portMAX_DELAY);
    }
}

// 消费者1：显示
void vDisplayTask(void *p)
{
    for(;;)
    {
        Data_t data;
        xQueuePeek(xDataQueue, &data, portMAX_DELAY);  // 只看不取
        显示数据(data);
        vTaskDelay(100);
    }
}

// 消费者2：存储
void vStorageTask(void *p)
{
    for(;;)
    {
        Data_t data;
        xQueueReceive(xDataQueue, &data, portMAX_DELAY);  // 真正取出
        存储数据(data);
    }
}
```

**注意**：这个模式需要仔细设计，可能需要多个队列或广播机制。

### 模式4：命令队列

```c
typedef enum
{
    CMD_LED_ON,
    CMD_LED_OFF,
    CMD_SET_BRIGHTNESS,
    CMD_RESET
} CommandId_t;

typedef struct
{
    CommandId_t eCommand;
    uint32_t ulParam;
} Command_t;

QueueHandle_t xCommandQueue;

// 发送命令的任务
void vUserInterfaceTask(void *p)
{
    for(;;)
    {
        if(按键按下())
        {
            Command_t cmd = {
                .eCommand = CMD_LED_ON,
                .ulParam = 0
            };
            xQueueSend(xCommandQueue, &cmd, 0);
        }
        vTaskDelay(10);
    }
}

// 执行命令的任务
void vCommandExecutorTask(void *p)
{
    for(;;)
    {
        Command_t cmd;
        xQueueReceive(xCommandQueue, &cmd, portMAX_DELAY);
        
        switch(cmd.eCommand)
        {
            case CMD_LED_ON:
                LED打开();
                break;
            case CMD_LED_OFF:
                LED关闭();
                break;
            case CMD_SET_BRIGHTNESS:
                设置亮度(cmd.ulParam);
                break;
            default:
                break;
        }
    }
}
```

**特点**：
- 使用枚举定义命令类型
- 结构体传递命令和参数
- 集中处理所有命令

## 数据传递技巧

### 技巧1：传递指针 vs 传递数据

#### 方式A：传递数据拷贝（推荐，安全）

```c
typedef struct {
    uint32_t id;
    char name[32];
    float value;
} SensorData_t;

// 队列存储整个结构体
QueueHandle_t xQueue = xQueueCreate(10, sizeof(SensorData_t));

// 发送数据
SensorData_t data = {.id = 1, .name = "Sensor1", .value = 25.5f};
xQueueSend(xQueue, &data, 0);  // 队列拷贝整个结构体

// 接收数据
SensorData_t received;
xQueueReceive(xQueue, &received, 0);  // 拷贝到接收缓冲区
```

**优点**：
- ✅ 线程安全，数据独立
- ✅ 发送后原数据可立即修改
- ✅ 不会有内存管理问题

**缺点**：
- ❌ 大数据结构体拷贝开销大
- ❌ 占用更多队列内存

---

#### 方式B：传递指针（高效，需小心）

```c
// 队列只存储指针
QueueHandle_t xQueue = xQueueCreate(10, sizeof(SensorData_t *));

// 动态分配内存
SensorData_t *pData = pvPortMalloc(sizeof(SensorData_t));
pData->id = 1;
strcpy(pData->name, "Sensor1");
pData->value = 25.5f;

// 发送指针
xQueueSend(xQueue, &pData, 0);  // 队列拷贝指针值

// 接收指针
SensorData_t *pReceived;
xQueueReceive(xQueue, &pReceived, 0);

// 使用数据
处理数据(pReceived);

// 记得释放！
vPortFree(pReceived);
```

**优点**：
- ✅ 只拷贝指针，效率高
- ✅ 队列占用内存小

**缺点**：
- ❌ 内存管理复杂，容易泄漏
- ❌ 需要确保数据生命周期
- ❌ 小心悬空指针

### 技巧2：使用结构体包装多种数据

```c
// 定义数据类型枚举
typedef enum
{
    TYPE_INTEGER,
    TYPE_FLOAT,
    TYPE_STRING,
    TYPE_SENSOR
} DataType_t;

// 定义联合体存储不同类型数据
typedef union
{
    uint32_t ulIntValue;
    float fFloatValue;
    char cStringValue[32];
    SensorData_t xSensorValue;
} DataUnion_t;

// 最终的消息结构体
typedef struct
{
    DataType_t eType;
    DataUnion_t xData;
} Message_t;

// 创建队列
QueueHandle_t xMessageQueue = xQueueCreate(10, sizeof(Message_t));

// 发送整数
void SendInteger(uint32_t value)
{
    Message_t msg;
    msg.eType = TYPE_INTEGER;
    msg.xData.ulIntValue = value;
    xQueueSend(xMessageQueue, &msg, 0);
}

// 发送字符串
void SendString(const char *str)
{
    Message_t msg;
    msg.eType = TYPE_STRING;
    strncpy(msg.xData.cStringValue, str, 31);
    msg.xData.cStringValue[31] = '\0';
    xQueueSend(xMessageQueue, &msg, 0);
}

// 接收并处理
void vMessageHandlerTask(void *p)
{
    for(;;)
    {
        Message_t msg;
        xQueueReceive(xMessageQueue, &msg, portMAX_DELAY);
        
        switch(msg.eType)
        {
            case TYPE_INTEGER:
                处理整数(msg.xData.ulIntValue);
                break;
            case TYPE_FLOAT:
                处理浮点数(msg.xData.fFloatValue);
                break;
            case TYPE_STRING:
                处理字符串(msg.xData.cStringValue);
                break;
            case TYPE_SENSOR:
                处理传感器数据(&msg.xData.xSensorValue);
                break;
        }
    }
}
```

### 技巧3：使用队列集合（Queue Sets）

```c
// 多个队列
QueueHandle_t xUARTQueue;
QueueHandle_t xSensorQueue;
QueueHandle_t xTimerQueue;

// 队列集合
QueueSetHandle_t xQueueSet;

// 初始化
void InitQueueSet(void)
{
    // 创建队列
    xUARTQueue = xQueueCreate(10, sizeof(uint8_t));
    xSensorQueue = xQueueCreate(5, sizeof(SensorData_t));
    xTimerQueue = xQueueCreate(3, sizeof(uint32_t));
    
    // 创建队列集合（能容纳所有队列的数据）
    xQueueSet = xQueueCreateSet(10 + 5 + 3);
    
    // 把队列添加到集合
    xQueueAddToSet(xUARTQueue, xQueueSet);
    xQueueAddToSet(xSensorQueue, xQueueSet);
    xQueueAddToSet(xTimerQueue, xQueueSet);
}

// 监控多个队列
void vMultiQueueMonitorTask(void *p)
{
    for(;;)
    {
        // 等待任何一个队列有数据
        QueueSetMemberHandle_t xActivatedMember;
        xActivatedMember = xQueueSelectFromSet(xQueueSet, portMAX_DELAY);
        
        // 判断是哪个队列
        if(xActivatedMember == xUARTQueue)
        {
            uint8_t data;
            xQueueReceive(xUARTQueue, &data, 0);
            处理UART数据(data);
        }
        else if(xActivatedMember == xSensorQueue)
        {
            SensorData_t data;
            xQueueReceive(xSensorQueue, &data, 0);
            处理传感器数据(&data);
        }
        else if(xActivatedMember == xTimerQueue)
        {
            uint32_t data;
            xQueueReceive(xTimerQueue, &data, 0);
            处理定时器事件(data);
        }
    }
}
```

## 队列与任务同步

### 同步模式1：使用队列作为信号量

```c
// 创建一个长度为1的队列
QueueHandle_t xSyncQueue = xQueueCreate(1, sizeof(uint8_t));

// 任务A：完成工作后通知
void vTaskA(void *p)
{
    for(;;)
    {
        做一些工作();
        
        // 发送一个"标记"表示完成
        uint8_t dummy = 0;
        xQueueSend(xSyncQueue, &dummy, 0);
    }
}

// 任务B：等待任务A完成
void vTaskB(void *p)
{
    for(;;)
    {
        uint8_t dummy;
        
        // 等待任务A完成
        xQueueReceive(xSyncQueue, &dummy, portMAX_DELAY);
        
        // 现在可以执行依赖任务A的工作
        执行后续工作();
    }
}
```

---

### 同步模式2：双向同步

```c
QueueHandle_t xRequestQueue;   // 请求队列
QueueHandle_t xResponseQueue;  // 响应队列

// 客户端任务
void vClientTask(void *p)
{
    for(;;)
    {
        // 发送请求
        Request_t req = 创建请求();
        xQueueSend(xRequestQueue, &req, portMAX_DELAY);
        
        // 等待响应
        Response_t resp;
        xQueueReceive(xResponseQueue, &resp, portMAX_DELAY);
        
        // 处理响应
        处理响应(&resp);
        
        vTaskDelay(1000);
    }
}

// 服务器任务
void vServerTask(void *p)
{
    for(;;)
    {
        // 等待请求
        Request_t req;
        xQueueReceive(xRequestQueue, &req, portMAX_DELAY);
        
        // 处理请求
        Response_t resp = 处理请求(&req);
        
        // 发送响应
        xQueueSend(xResponseQueue, &resp, portMAX_DELAY);
    }
}
```

## 实际应用案例

### 案例1：串口数据接收

```c
// 场景：中断接收串口数据，任务处理数据

QueueHandle_t xUARTDataQueue;

// 串口中断（简化版）
void UART_IRQHandler(void)
{
    BaseType_t xHigherPriorityTaskWoken = pdFALSE;
    
    if(有数据到达())
    {
        uint8_t data = 读取数据();
        
        // 在中断中使用 FromISR 版本！
        xQueueSendFromISR(xUARTDataQueue, &data, &xHigherPriorityTaskWoken);
        
        // 如果唤醒了更高优先级任务，触发上下文切换
        portYIELD_FROM_ISR(xHigherPriorityTaskWoken);
    }
}

// 数据处理任务
void vUARTDataProcessTask(void *p)
{
    for(;;)
    {
        uint8_t data;
        
        // 等待中断送来的数据
        xQueueReceive(xUARTDataQueue, &data, portMAX_DELAY);
        
        // 在这里做耗时的处理
        解析数据(data);
        存储数据(data);
        显示数据(data);
    }
}
```

**关键点**：
- 中断中必须使用 `FromISR` 版本的函数
- 中断只做最必要的工作（接收数据）
- 复杂处理留给任务

### 案例2：传感器数据采集系统

```c
typedef struct
{
    uint32_t ulSensorId;
    uint32_t ulTimestamp;
    float fTemperature;
    float fHumidity;
    float fPressure;
} SensorReading_t;

// 数据队列
QueueHandle_t xSensorDataQueue;

// 传感器采集任务（高优先级）
void vSensorCollectTask(void *p)
{
    for(;;)
    {
        // 定时采集
        vTaskDelayUntil(&xLastWakeTime, pdMS_TO_TICKS(100));
        
        SensorReading_t reading;
        reading.ulSensorId = 1;
        reading.ulTimestamp = 获取时间戳();
        reading.fTemperature = 读取温度();
        reading.fHumidity = 读取湿度();
        reading.fPressure = 读取气压();
        
        // 发送到队列（如果队列满，丢弃旧数据）
        if(xQueueSend(xSensorDataQueue, &reading, 0) != pdPASS)
        {
            // 队列满，移除旧数据再试
            SensorReading_t dummy;
            xQueueReceive(xSensorDataQueue, &dummy, 0);
            xQueueSend(xSensorDataQueue, &reading, 0);
        }
    }
}

// 数据处理任务（中等优先级）
void vDataProcessTask(void *p)
{
    for(;;)
    {
        SensorReading_t reading;
        xQueueReceive(xSensorDataQueue, &reading, pdMS_TO_TICKS(500));
        
        if(有新数据(&reading))
        {
            滤波处理(&reading);
            计算统计值(&reading);
            检查报警(&reading);
        }
    }
}

// 显示任务（低优先级）
void vDisplayTask(void *p)
{
    for(;;)
    {
        vTaskDelay(pdMS_TO_TICKS(500));
        
        // Peek最新数据（不移除）
        SensorReading_t latest;
        if(xQueuePeek(xSensorDataQueue, &latest, 0) == pdPASS)
        {
            更新显示(&latest);
        }
    }
}
```

## 性能优化建议

### 1. 合理选择队列长度

| 场景 | 推荐长度 | 理由 |
|------|---------|------|
| 简单通知 | 1-2 | 只需要同步，不需要缓冲 |
| 常规数据 | 5-10 | 平衡内存和性能 |
| 突发数据 | 20-50 | 根据最大突发量确定 |
| 高速数据流 | 50-100+ | 防止数据丢失 |

---

### 2. 优化数据大小

```c
// ❌ 不好：太大的结构体
typedef struct {
    uint8_t data[1024];  // 1KB！
} BigData_t;
xQueueCreate(10, sizeof(BigData_t));  // 需要10KB！

// ✅ 好：传递指针
typedef struct {
    uint8_t data[1024];
} BigData_t;
xQueueCreate(10, sizeof(BigData_t *));  // 只需要10×4=40字节！
```

---

### 3. 合理设置阻塞时间

```c
// ❌ 不好：无限等待可能导致死锁
xQueueSend(xQueue, &data, portMAX_DELAY);

// ✅ 好：设置超时，添加错误处理
if(xQueueSend(xQueue, &data, pdMS_TO_TICKS(100)) != pdPASS)
{
    // 超时处理
   记录错误("队列发送超时");
}
```

---

### 4. 使用查询函数优化

```c
// 发送前先检查是否有空间
if(uxQueueSpacesAvailable(xQueue) > 0)
{
    xQueueSend(xQueue, &data, 0);  // 有空间才发送，不阻塞
}

// 接收前先检查是否有数据
if(uxQueueMessagesWaiting(xQueue) > 0)
{
    xQueueReceive(xQueue, &data, 0);  // 有数据才接收
}
```
